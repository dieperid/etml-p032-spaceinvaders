147  // Boucle foreach pour lire tout le contenu du fichier texte et trouver un texte qui contient le pseudo du joueur
148  foreach (var match in File.ReadLines(_pathFile).Select((text, index) => new { text, lineNumber = index + 1 }).Where(x => x.text.Contains(Pseudo)))
149  {

151  arrayContent = match.text.Split(' ');
152 
153  // Si la case 0 du tableau correspond à un pseudo
154  if (arrayContent[0] == Pseudo)
155  {
156  fileResult = match.text;
157  }
158  }
159 
160  // Si la variable fileResult n'est pas vide
161  if (fileResult != "")
162  {
163  // Recherche seuelement des nombre dans le contenu récupéré
164  MatchCollection matches = Regex.Matches(fileResult, "[0-9]");
165  foreach (Match match in matches)
166  {
167  lastScore += match.Value;
168  }
169 
170  // On converti l'ancien score
171  fileContent = File.ReadAllText(_pathFile);
172 
173  // Si le score qu'il vient de faire est supérieur à l'ancien
174  if (state.GameScore > Convert.ToInt32(lastScore))
175  {
176  // On remplace son score dans la variable qui contient le contenu du fichier texte
177  newFileContent = fileContent.Replace(Pseudo + " \t: " + lastScore, Pseudo + " \t: " + state.GameScore);
178 
179  // On écrit dans le fichier texte les nouvelles données
180  using (FileStream fs = File.Create(_pathFile))
181  {
182  Byte[] textToWrite = new UTF8Encoding(true).GetBytes(newFileContent);
183  fs.Write(textToWrite, 0, textToWrite.Length);
